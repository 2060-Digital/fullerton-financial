import { Meta, Typeset } from "@storybook/addon-docs/blocks"

import resolveConfig from "tailwindcss/resolveConfig"
import tailwindConfig from "tailwind.config.js"

<Meta title="Typography" />

export const typography = {
  type: {
    primary: resolveConfig(tailwindConfig).theme.fontFamily.mono,
  },
  weight: {
    light: "300",
    regular: "400",
    medium: "500",
  },
  sizes: Object.values(resolveConfig(tailwindConfig).theme.fontSize).map((s) => {
    const value = typeof s === "string" ? s : s[0]
    return value.split("rem")[0] * getComputedStyle(document.documentElement).fontSize.split("px")[0] + "px"
  }),
}

export const SampleText = "Lorem ipsum dolor sit amet, consectetur adipiscing elit."
