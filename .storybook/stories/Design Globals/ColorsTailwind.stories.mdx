import resolveConfig from "tailwindcss/resolveConfig"
import tailwindConfig from "tailwind.config.js"

# All Available Tailwind Colors

We don't have to use these. In fact, in some cases it may be a good idea not to use these. However, in the case that we might want to do so, here is a list of all available Tailwind colors from the default theme.

<ColorPalette>
  {Object.entries(resolveConfig(tailwindConfig).theme.colors).map(([k, v]) => {
    if (typeof v === "string" && v.match(/^#/)) {
      return <ColorItem key={k} title={k.toUpperCase()} colors={{ [k]: v }} />
    }
    if (typeof v === "object") {
      let customColors = Object.fromEntries(Object.entries(v))
      if (Object.keys(customColors).length > 0) {
        return <ColorItem key={k} title={k.toUpperCase()} colors={customColors} />
      }
    }
    return null
  })}
</ColorPalette>
